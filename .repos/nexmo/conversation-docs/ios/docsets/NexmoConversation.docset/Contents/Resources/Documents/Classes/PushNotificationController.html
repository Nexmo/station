AppDelegate Support
  • registeredForRemoteNotifications(with:)

    Register for remote notifications with device token. Call this method from your AppDelegate.

    • Example: Swift
    func application(_ application: UIApplication, didRegisterForRemoteNotificationsWithDeviceToken deviceToken: Data) {
   // push device token to SDK
   client.appLifecycle.push.registeredForRemoteNotifications(with: deviceToken)
}
    • Example: Objective-C
    - (void)application:(UIApplication *)application didRegisterForRemoteNotificationsWithDeviceToken:(NSData *)deviceToken {
   // push device token to SDK
   [client.push registeredForRemoteNotificationsWith: deviceToken];
}
